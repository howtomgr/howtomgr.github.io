Apache HTTP Server Installation Guide

World's most popular web server software serving over 40% of active websites. Reliable, secure, and highly configurable for hosting websites and web applications.

Prerequisites

  • Linux system (any modern distribution)
  • Root or sudo access
  • 1GB RAM minimum, 2GB+ recommended for production
  • Network connectivity for package downloads

    • Installation

    Ubuntu/Debian

    bash 
    # Update package list
sudo apt update

# Install Apache and essential modules
sudo apt install -y apache2 apache2-utils

# Enable essential modules
sudo a2enmod rewrite ssl headers deflate expires security2

# Start and enable Apache
sudo systemctl enable --now apache2

# Check status
sudo systemctl status apache2

    RHEL/CentOS/Rocky Linux/AlmaLinux

    bash 
    # Install Apache (httpd)
sudo yum install -y httpd httpd-tools mod_ssl

# For newer versions with dnf
sudo dnf install -y httpd httpd-tools mod_ssl

# Start and enable Apache
sudo systemctl enable --now httpd

# Check status
sudo systemctl status httpd

    Fedora

      - webserver

  webserver:
    driver: bridge

docker-compose up -d

    Source Compilation (Advanced)

    bash 
    # Install build dependencies
# Ubuntu/Debian
sudo apt install -y build-essential libapr1-dev libaprutil1-dev libpcre3-dev libssl-dev zlib1g-dev

# RHEL/CentOS
sudo yum groupinstall -y "Development Tools"
sudo yum install -y apr-devel apr-util-devel pcre-devel openssl-devel zlib-devel

# Download Apache source
cd /tmp
wget https://archive.apache.org/dist/httpd/httpd-2.4.57.tar.gz
tar xzf httpd-2.4.57.tar.gz
cd httpd-2.4.57

# Configure with security and performance modules
./configure \
  --prefix=/etc/apache2 \
  --exec-prefix=/usr \
  --bindir=/usr/bin \
  --sbindir=/usr/sbin \
  --sysconfdir=/etc/apache2 \
  --libexecdir=/usr/lib/apache2/modules \
  --localstatedir=/var \
  --datadir=/usr/share/apache2 \
  --includedir=/usr/include/apache2 \
  --infodir=/usr/share/info \
  --mandir=/usr/share/man \
  --with-program-name=apache2 \
  --enable-layout=Debian \
  --with-regex=pcre \
  --enable-mods-shared=reallyall \
  --enable-ssl \
  --enable-rewrite \
  --enable-headers \
  --enable-deflate \
  --enable-expires \
  --enable-security \
  --with-ssl=/usr \
  --with-ldap \
  --with-ldap-include=/usr/include \
  --with-ldap-lib=/usr/lib

# Compile and install
make -j$(nproc)
sudo make install

# Create apache user
sudo useradd -r -s /sbin/nologin apache

# Create systemd service
sudo tee /etc/systemd/system/apache2.service > /dev/null <<EOF
[Unit]
Description=The Apache HTTP Server
After=network.target remote-fs.target nss-lookup.target

[Service]
Type=notify
ExecStart=/usr/sbin/apache2 -DFOREGROUND
ExecReload=/usr/sbin/apache2ctl graceful
ExecStop=/bin/kill -WINTERM \${MAINPID}
KillSignal=SIGCONT
PrivateTmp=true

[Install]
WantedBy=multi-user.target
EOF

sudo systemctl daemon-reload
sudo systemctl enable --now apache2

    Security Hardening (2024 Best Practices)

    Hide Server Information

    bash 
    # Create security configuration
# Ubuntu/Debian
sudo tee /etc/apache2/conf-available/security-hardening.conf > /dev/null <<EOF
# Hide Apache version and OS information
ServerTokens Prod
ServerSignature Off

# Hide PHP version (if using PHP)
Header always unset X-Powered-By
Header unset X-Powered-By

# Security headers (2024 recommendations)
Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains; preload"
Header always set X-Frame-Options "SAMEORIGIN"
Header always set X-Content-Type-Options "nosniff"
Header always set X-XSS-Protection "1; mode=block"
Header always set Referrer-Policy "strict-origin-when-cross-origin"
Header always set Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline'; img-src 'self' data: https:; font-src 'self' data:; connect-src 'self'; media-src 'self'; object-src 'none'; child-src 'self'; form-action 'self'; base-uri 'self';"
Header always set Permissions-Policy "camera=(), microphone=(), geolocation=()"

# Remove sensitive headers
Header always unset Server
Header unset Server

# Disable server status and info
<Location "/server-status">
    SetHandler server-status
    Require local
</Location>

<Location "/server-info">
    SetHandler server-info
    Require local
</Location>

# Disable trace method
TraceEnable off

# Timeout configurations
Timeout 60
KeepAliveTimeout 15

# Limit request size (adjust as needed)
LimitRequestBody 10485760  # 10MB

# Hide .htaccess files
<FilesMatch "^\.ht">
    Require all denied
</FilesMatch>

# Hide backup and temporary files
<FilesMatch "\.(bak|backup|swp|tmp|~)$">
    Require all denied
</FilesMatch>

# Disable directory browsing by default
Options -Indexes

# Information disclosure protection
<DirectoryMatch "/\.">
    Require all denied
</DirectoryMatch>

# Clickjacking protection
Header always append X-Frame-Options SAMEORIGIN
EOF

# Enable security configuration
sudo a2enconf security-hardening
sudo systemctl reload apache2

# RHEL/CentOS (add to main configuration)
sudo tee -a /etc/httpd/conf/httpd.conf > /dev/null <<EOF

# Security hardening
ServerTokens Prod
ServerSignature Off

# Load security headers module
LoadModule headers_module modules/mod_headers.so

# Security headers
Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains; preload"
Header always set X-Frame-Options "SAMEORIGIN"
Header always set X-Content-Type-Options "nosniff"
Header always set X-XSS-Protection "1; mode=block"
Header always set Referrer-Policy "strict-origin-when-cross-origin"

# Disable trace method
TraceEnable off

# Hide .htaccess files
<FilesMatch "^\.ht">
    Require all denied
</FilesMatch>

# Disable directory browsing
Options -Indexes
EOF

sudo systemctl reload httpd

    ModSecurity WAF Installation

    bash 
    # Ubuntu/Debian
sudo apt install -y libapache2-mod-security2

# RHEL/CentOS
sudo yum install -y mod_security mod_security_crs

# Enable ModSecurity
sudo a2enmod security2  # Ubuntu/Debian

# Download OWASP Core Rule Set
sudo mkdir -p /etc/apache2/modsecurity.d
cd /etc/apache2/modsecurity.d
sudo wget https://github.com/coreruleset/coreruleset/archive/refs/tags/v3.3.4.tar.gz
sudo tar xzf v3.3.4.tar.gz
sudo mv coreruleset-3.3.4 crs

# Configure ModSecurity
sudo tee /etc/apache2/mods-available/security2.conf > /dev/null <<EOF
<IfModule mod_security2.c>
    SecRuleEngine On
    SecRequestBodyAccess On
    SecRequestBodyLimit 13107200
    SecRequestBodyNoFilesLimit 131072
    SecRequestBodyInMemoryLimit 131072
    SecRequestBodyLimitAction Reject
    SecRule REQUEST_HEADERS:Content-Type "text/xml" \
         "id:'200000',phase:1,t:none,t:lowercase,pass,nolog,ctl:requestBodyProcessor=XML"
    SecRule REQUEST_HEADERS:Content-Type "application/json" \
         "id:'200001',phase:1,t:none,t:lowercase,pass,nolog,ctl:requestBodyProcessor=JSON"
    SecRule REQBODY_ERROR "!@eq 0" \
    "id:'200002', phase:2,t:none,log,deny,status:400,msg:'Failed to parse request body.',logdata:'Error %{REQBODY_ERROR_MSG}',severity:2"
    SecRule MULTIPART_STRICT_ERROR "!@eq 0" \
    "id:'200003',phase:2,t:none,log,deny,status:400, \
    msg:'Multipart request body failed strict validation: \
    PE %{REQBODY_PROCESSOR_ERROR}, \
    BQ %{MULTIPART_BOUNDARY_QUOTED}, \
    BW %{MULTIPART_BOUNDARY_WHITESPACE}, \
    DB %{MULTIPART_DATA_BEFORE}, \
    DA %{MULTIPART_DATA_AFTER}, \
    HF %{MULTIPART_HEADER_FOLDING}, \
    LF %{MULTIPART_LF_LINE}, \
    SM %{MULTIPART_MISSING_SEMICOLON}, \
    IQ %{MULTIPART_INVALID_QUOTING}, \
    IP %{MULTIPART_INVALID_PART}, \
    IH %{MULTIPART_INVALID_HEADER_FOLDING}, \
    FL %{MULTIPART_FILE_LIMIT_EXCEEDED}'"

    SecRule MULTIPART_UNMATCHED_BOUNDARY "!@eq 0" \
    "id:'200004',phase:2,t:none,log,deny,msg:'Multipart parser detected a possible unmatched boundary.'"

    SecPcreMatchLimit 1000
    SecPcreMatchLimitRecursion 1000

    SecRule TX:/^MSC_/ "!@streq 0" \
            "id:'200005',phase:2,t:none,deny,msg:'ModSecurity internal error flagged: %{MATCHED_VAR_NAME}'"

    SecResponseBodyAccess Off
    SecDebugLog /var/log/apache2/modsec_debug.log
    SecDebugLogLevel 0
    SecAuditEngine RelevantOnly
    SecAuditLogRelevantStatus "^(?:5|4(?!04))"
    SecAuditLogParts ABDEFHIJZ
    SecAuditLogType Serial
    SecAuditLog /var/log/apache2/modsec_audit.log
    
    # Include OWASP Core Rule Set
    Include /etc/apache2/modsecurity.d/crs/crs-setup.conf.example
    Include /etc/apache2/modsecurity.d/crs/rules/*.conf
</IfModule>
EOF

# Create data directory
sudo mkdir -p /var/cache/modsecurity
sudo chown www-data:www-data /var/cache/modsecurity

sudo systemctl reload apache2

    SSL/TLS Configuration (Modern 2024 Standards)

    bash 
    # Create modern SSL configuration
sudo tee /etc/apache2/conf-available/ssl-modern.conf > /dev/null <<EOF
# Modern SSL configuration (2024)
SSLEngine on

# Protocols - Only TLS 1.2 and 1.3
SSLProtocol all -SSLv3 -TLSv1 -TLSv1.1

# Cipher suites for modern browsers (2024)
SSLCipherSuite ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384

# Prefer server ciphers for TLS 1.2, client ciphers for TLS 1.3
SSLHonorCipherOrder off

# HSTS (HTTP Strict Transport Security)
Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains; preload"

# Session cache
SSLSessionCache shmcb:/var/run/ssl_scache(512000)
SSLSessionCacheTimeout 300

# OCSP Stapling
SSLUseStapling On
SSLStaplingCache shmcb:/var/run/ocsp(128000)
SSLStaplingResponderTimeout 5
SSLStaplingReturnResponderErrors off

# Modern security options
SSLCompression off
SSLSessionTickets off

# Perfect Forward Secrecy
SSLOpenSSLConfCmd DHParameters /etc/ssl/certs/dhparam.pem
EOF

# Generate strong DH parameters
sudo openssl dhparam -out /etc/ssl/certs/dhparam.pem 2048

# Enable SSL configuration
sudo a2enconf ssl-modern
sudo a2enmod ssl headers
sudo systemctl reload apache2

    Secure Virtual Host Configuration

    bash 
    # Create production-ready virtual host
sudo tee /etc/apache2/sites-available/secure-site.conf > /dev/null <<EOF
# HTTP to HTTPS redirect
<VirtualHost *:80>
    ServerName example.com
    ServerAlias www.example.com
    
    # Security headers for redirects
    Header always set X-Content-Type-Options nosniff
    Header always set X-Frame-Options DENY
    
    # Redirect to HTTPS
    Redirect permanent / https://example.com/
</VirtualHost>

# HTTPS Virtual Host
<VirtualHost *:443>
    ServerName example.com
    ServerAlias www.example.com
    DocumentRoot /var/www/html/example.com
    
    # SSL Configuration
    SSLEngine on
    SSLCertificateFile /etc/letsencrypt/live/example.com/fullchain.pem
    SSLCertificateKeyFile /etc/letsencrypt/live/example.com/privkey.pem
    
    # Security headers (2024 standards)
    Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains; preload"
    Header always set X-Frame-Options "SAMEORIGIN"
    Header always set X-Content-Type-Options "nosniff"
    Header always set X-XSS-Protection "1; mode=block"
    Header always set Referrer-Policy "strict-origin-when-cross-origin"
    Header always set Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline'; img-src 'self' data: https:; font-src 'self' data:; connect-src 'self'; media-src 'self'; object-src 'none'; child-src 'self'; form-action 'self'; base-uri 'self';"
    Header always set Permissions-Policy "camera=(), microphone=(), geolocation=()"
    
    # Remove server information
    Header always unset Server
    Header unset X-Powered-By
    
    <Directory /var/www/html/example.com>
        Options -Indexes +FollowSymLinks
        AllowOverride All
        Require all granted
        
        # Additional security
        <Files ".*">
            Require all denied
        </Files>
        
        <Files "*.php~">
            Require all denied
        </Files>
        
        <Files "*.inc">
            Require all denied
        </Files>
    </Directory>
    
    # PHP configuration (if using PHP)
    <IfModule mod_php.c>
        php_admin_flag allow_url_include Off
        php_admin_flag allow_url_fopen Off
        php_admin_flag file_uploads On
        php_admin_value upload_max_filesize 64M
        php_admin_value post_max_size 64M
        php_admin_value memory_limit 256M
        php_admin_value max_execution_time 300
        php_admin_flag expose_php Off
    </IfModule>
    
    # Compression
    <IfModule mod_deflate.c>
        SetOutputFilter DEFLATE
        AddOutputFilterByType DEFLATE text/html text/css text/javascript text/xml text/plain application/javascript application/xml+rss application/xml application/x-javascript application/json
        
        # Don't compress images
        SetEnvIfNoCase Request_URI \.(?:gif|jpe?g|png)$ no-gzip dont-vary
        SetEnvIfNoCase Request_URI \.(?:exe|t?gz|zip|bz2|sit|rar)$ no-gzip dont-vary
    </IfModule>
    
    # Caching rules
    <IfModule mod_expires.c>
        ExpiresActive On
        ExpiresByType image/jpg "access plus 1 month"
        ExpiresByType image/jpeg "access plus 1 month"
        ExpiresByType image/gif "access plus 1 month"
        ExpiresByType image/png "access plus 1 month"
        ExpiresByType image/webp "access plus 1 month"
        ExpiresByType text/css "access plus 1 month"
        ExpiresByType application/pdf "access plus 1 month"
        ExpiresByType text/javascript "access plus 1 month"
        ExpiresByType application/javascript "access plus 1 month"
        ExpiresByType application/x-javascript "access plus 1 month"
        ExpiresByType application/x-shockwave-flash "access plus 1 month"
        ExpiresByType image/x-icon "access plus 1 year"
        ExpiresDefault "access plus 2 days"
    </IfModule>
    
    # Rate limiting (if mod_evasive is available)
    <IfModule mod_evasive24.c>
        DOSHashTableSize    1024
        DOSPageCount        2
        DOSSiteCount        50
        DOSPageInterval     1
        DOSSiteInterval     1
        DOSBlockingPeriod   600
        DOSLogDir           /var/log/apache2/
        DOSEmailNotify      admin@example.com
    </IfModule>
    
    # ModSecurity
    <IfModule mod_security2.c>
        SecRuleEngine On
        SecDefaultAction "log,auditlog,pass"
    </IfModule>
    
    ErrorLog \${APACHE_LOG_DIR}/example.com_error.log
    CustomLog \${APACHE_LOG_DIR}/example.com_access.log combined
    
    # Log security events
    CustomLog \${APACHE_LOG_DIR}/example.com_security.log "%h %l %u %t \"%r\" %>s %b \"%{Referer}i\" \"%{User-agent}i\"" env=security
</VirtualHost>

# WWW redirect
<VirtualHost *:443>
    ServerName www.example.com
    
    SSLEngine on
    SSLCertificateFile /etc/letsencrypt/live/example.com/fullchain.pem
    SSLCertificateKeyFile /etc/letsencrypt/live/example.com/privkey.pem
    
    Redirect permanent / https://example.com/
</VirtualHost>
EOF

sudo a2ensite secure-site.conf
sudo a2enconf security-hardening
sudo systemctl reload apache2

    ExpiresByType image/gif "access plus 1 year"
    ExpiresByType image/png "access plus 1 year"
    ExpiresByType image/webp "access plus 1 year"
    ExpiresByType text/css "access plus 1 month"
    ExpiresByType application/pdf "access plus 1 month"
    ExpiresByType text/javascript "access plus 1 month"
    ExpiresByType application/javascript "access plus 1 month"
    ExpiresByType application/x-javascript "access plus 1 month"
    ExpiresByType application/x-shockwave-flash "access plus 1 month"
    ExpiresByType image/x-icon "access plus 1 year"
    ExpiresByType text/html "access plus 300 seconds"
    
    # Default expiration
    ExpiresDefault "access plus 1 day"
</IfModule>

# Compression
<IfModule mod_deflate.c>
    # Enable compression for text files
    <Location />
        SetOutputFilter DEFLATE
        
        # Don't compress images or binaries
        SetEnvIfNoCase Request_URI \
            \.(?:gif|jpe?g|png|webp)$ no-gzip dont-vary
        SetEnvIfNoCase Request_URI \
            \.(?:exe|t?gz|zip|bz2|sit|rar|pdf)$ no-gzip dont-vary
            
        # Include file types to compress
        AddOutputFilterByType DEFLATE text/html text/plain text/xml text/css text/javascript
        AddOutputFilterByType DEFLATE application/javascript application/json application/xml application/xhtml+xml application/rss+xml
    </Location>
</IfModule>
EOF

    Backup and Maintenance

    Automated Backup Script

    bash 
    sudo tee /usr/local/bin/apache-backup.sh > /dev/null <<'EOF'
#!/bin/bash
BACKUP_DIR="/backup/apache"
DATE=$(date +%Y%m%d_%H%M%S)

mkdir -p ${BACKUP_DIR}/{config,logs,sites}

# Detect distribution and set paths
if [ -d /etc/apache2 ]; then
    # Ubuntu/Debian
    CONFIG_DIR="/etc/apache2"
    LOG_DIR="/var/log/apache2"
    SERVICE_NAME="apache2"
elif [ -d /etc/httpd ]; then
    # RHEL/CentOS
    CONFIG_DIR="/etc/httpd"
    LOG_DIR="/var/log/httpd"  
    SERVICE_NAME="httpd"
fi

# Backup configuration
tar -czf ${BACKUP_DIR}/config/apache-config-${DATE}.tar.gz -C / ${CONFIG_DIR#/}

# Backup website files
tar -czf ${BACKUP_DIR}/sites/apache-sites-${DATE}.tar.gz -C /var/www .

# Backup recent logs (last 7 days)
find ${LOG_DIR} -name "*.log" -mtime -7 -exec tar -czf ${BACKUP_DIR}/logs/apache-logs-${DATE}.tar.gz {} +

# Keep only last 7 backups
find ${BACKUP_DIR} -name "apache-*" -type f -mtime +7 -delete

echo "Apache backup completed: ${DATE}"
EOF

sudo chmod +x /usr/local/bin/apache-backup.sh

# Schedule daily backup
echo "0 2 * * * root /usr/local/bin/apache-backup.sh" | sudo tee -a /etc/crontab

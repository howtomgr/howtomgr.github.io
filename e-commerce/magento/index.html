1. Prerequisites

  • Hardware Requirements:
  • CPU: 4+ cores
  • RAM: 4GB minimum
  • Storage: 20GB for data
  • Network: HTTP/HTTPS
  • Operating System:
  • Linux: Any modern distribution (RHEL, Debian, Ubuntu, CentOS, Fedora, Arch, Alpine, openSUSE)
  • macOS: 10.14+ (Mojave or newer)
  • Windows: Windows Server 2016+ or Windows 10
  • FreeBSD: 11.0+
  • Network Requirements:
  • Port 80 (default magento port)
  • None
  • Dependencies:
  • See official documentation for specific requirements
  • System Access: root or sudo privileges required

    • 2. Supported Operating Systems

    This guide supports installation on:

  • RHEL 8/9 and derivatives (CentOS Stream, Rocky Linux, AlmaLinux)
  • Debian 11/12
  • Ubuntu 20.04/22.04/24.04 LTS
  • Arch Linux (rolling release)
  • Alpine Linux 3.18+
  • openSUSE Leap 15.5+ / Tumbleweed
  • SUSE Linux Enterprise Server (SLES) 15+
  • macOS 12+ (Monterey and later)
  • FreeBSD 13+
  • Windows 10/11/Server 2019+ (where applicable)

    • 3. Installation

    RHEL/CentOS/Rocky Linux/AlmaLinux

    bash 
    # Install EPEL repository if needed
sudo dnf install -y epel-release

# Install magento
sudo dnf install -y magento

# Enable and start service
sudo systemctl enable --now magento

# Configure firewall
sudo firewall-cmd --permanent --add-port=80/tcp
sudo firewall-cmd --reload

# Verify installation
magento --version

    Debian/Ubuntu

    bash 
    # Update package index
sudo apt update

# Install magento
sudo apt install -y magento

# Enable and start service
sudo systemctl enable --now magento

# Configure firewall
sudo ufw allow 80

# Verify installation
magento --version

    Arch Linux

    bash 
    # Install magento
sudo pacman -S magento

# Enable and start service
sudo systemctl enable --now magento

# Verify installation
magento --version

    Alpine Linux

    bash 
    # Install magento
apk add --no-cache magento

# Enable and start service
rc-update add magento default
rc-service magento start

# Verify installation
magento --version

    openSUSE/SLES

    bash 
    # Install magento
sudo zypper install -y magento

# Enable and start service
sudo systemctl enable --now magento

# Configure firewall
sudo firewall-cmd --permanent --add-port=80/tcp
sudo firewall-cmd --reload

# Verify installation
magento --version

    macOS

    bash 
    # Using Homebrew

    5. Service Management

    systemd (RHEL, Debian, Ubuntu, Arch, openSUSE)

    bash 
    # Enable service
sudo systemctl enable magento

# Start service
sudo systemctl start magento

# Stop service
sudo systemctl stop magento

# Restart service
sudo systemctl restart magento

# Check status
sudo systemctl status magento

# View logs
sudo journalctl -u magento -f

    OpenRC (Alpine Linux)

    bash 
    # Enable service
rc-update add magento default

# Start service
rc-service magento start

# Stop service
rc-service magento stop

# Restart service
rc-service magento restart

# Check status
rc-service magento status

    rc.d (FreeBSD)

    bash 
    # Enable in /etc/rc.conf
echo 'magento_enable="YES"' >> /etc/rc.conf

# Start service
service magento start

# Stop service
service magento stop

# Restart service
service magento restart

# Check status
service magento status

    Reverse Proxy Setup

    nginx Configuration

    nginx 
    upstream magento_backend {
    server 127.0.0.1:80;
}

server {
    listen 80;
    server_name magento.example.com;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name magento.example.com;

    ssl_certificate /etc/ssl/certs/magento.example.com.crt;
    ssl_certificate_key /etc/ssl/private/magento.example.com.key;

    location / {
        proxy_pass http://magento_backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

    Apache Configuration

    apache 
    <VirtualHost *:80>
    ServerName magento.example.com
    Redirect permanent / https://magento.example.com/
</VirtualHost>

<VirtualHost *:443>
    ServerName magento.example.com
    
    SSLEngine on
    SSLCertificateFile /etc/ssl/certs/magento.example.com.crt
    SSLCertificateKeyFile /etc/ssl/private/magento.example.com.key
    
    ProxyRequests Off
    ProxyPreserveHost On
    
    ProxyPass / http://127.0.0.1:80/
    ProxyPassReverse / http://127.0.0.1:80/
</VirtualHost>

    HAProxy Configuration

    haproxy 
    frontend magento_frontend
    bind *:80
    bind *:443 ssl crt /etc/ssl/certs/magento.pem
    redirect scheme https if !{ ssl_fc }
    default_backend magento_backend

backend magento_backend
    balance roundrobin
    server magento1 127.0.0.1:80 check

    Basic Security Setup

    bash

# Configure firewall
sudo firewall-cmd --permanent --add-port=80/tcp
sudo firewall-cmd --reload

# Enable SELinux policies (if applicable)
sudo setsebool -P httpd_can_network_connect on

    9. Backup and Restore

    Backup Script

    bash 
    #!/bin/bash
# Basic backup script
BACKUP_DIR="/backup/magento"
DATE=$(date +%Y%m%d_%H%M%S)

mkdir -p "$BACKUP_DIR"
tar -czf "$BACKUP_DIR/magento-backup-$DATE.tar.gz" /etc/magento /var/lib/magento

echo "Backup completed: $BACKUP_DIR/magento-backup-$DATE.tar.gz"

    Restore Procedure

    bash 
    # Stop service
sudo systemctl stop magento

# Restore from backup
tar -xzf /backup/magento/magento-backup-*.tar.gz -C /

# Start service
sudo systemctl start magento

    6. Troubleshooting

    Common Issues

    1. Service won't start:

    bash 
    # Check logs
sudo journalctl -u magento -n 100
sudo tail -f /var/log/magento/magento.log

# Check configuration
magento --version

# Check permissions
ls -la /etc/magento

    2. Connection issues:

    bash 
    # Check if service is listening
sudo ss -tlnp | grep 80

# Test connectivity
telnet localhost 80

# Check firewall
sudo firewall-cmd --list-all

    3. Performance issues:

    bash 
    # Check resource usage
top -p $(pgrep magento)

# Check disk I/O
iotop -p $(pgrep magento)

# Check connections
ss -an | grep 80

    Maintenance

    Update Procedures

    bash 
    # RHEL/CentOS/Rocky/AlmaLinux
sudo dnf update magento

# Debian/Ubuntu
sudo apt update && sudo apt upgrade magento

# Arch Linux
sudo pacman -Syu magento

# Alpine Linux
apk update && apk upgrade magento

# openSUSE
sudo zypper update magento

# FreeBSD
pkg update && pkg upgrade magento

# Always backup before updates
tar -czf /backup/magento-pre-update-$(date +%Y%m%d).tar.gz /etc/magento

# Restart after updates
sudo systemctl restart magento

    Regular Maintenance

    bash 
    # Log rotation
sudo logrotate -f /etc/logrotate.d/magento

# Clean old logs
find /var/log/magento -name "*.log" -mtime +30 -delete

# Check disk usage
du -sh /var/lib/magento

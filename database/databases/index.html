Database Installation Guide

Installation instructions for popular database systems including MySQL/MariaDB, PostgreSQL, MongoDB, and more.

Prerequisites

  • RHEL/CentOS 7+ or compatible Linux distribution
  • Root or sudo access
  • Sufficient disk space for database storage

    • MariaDB/MySQL

    Installation

    bash 
    # Add MariaDB repository
echo '[mariadb]
name = MariaDB
baseurl = https://mirror.its.dal.ca/mariadb/yum/10.6/centos$releasever-$basearch
module_hotfixes=1
gpgkey=https://mirror.its.dal.ca/mariadb/yum/RPM-GPG-KEY-MariaDB
gpgcheck=1' | sudo tee /etc/yum.repos.d/MariaDB.repo

# Install MariaDB
sudo yum install -y mariadb-server mariadb
sudo systemctl enable --now mariadb

# Secure installation
sudo mysql_secure_installation

    PostgreSQL

    Installation

    bash 
    # Install PostgreSQL repository
sudo yum install -y https://download.postgresql.org/pub/repos/yum/reporpms/EL-7-x86_64/pgdg-redhat-repo-latest.noarch.rpm

# Install PostgreSQL
sudo yum install -y postgresql13-server postgresql13
sudo /usr/pgsql-13/bin/postgresql-13-setup initdb
sudo systemctl enable --now postgresql-13

# Configure authentication
sudo -u postgres psql
postgres=# CREATE USER myuser WITH PASSWORD 'mypassword';
postgres=# CREATE DATABASE mydb OWNER myuser;
postgres=# \q

    MongoDB

    Installation

    bash 
    # Add MongoDB repository
echo '[mongodb-org-5.0]
name=MongoDB Repository
baseurl=https://repo.mongodb.org/yum/redhat/$releasever/mongodb-org/5.0/$basearch/
gpgcheck=1
enabled=1
gpgkey=https://www.mongodb.org/static/pgp/server-5.0.asc' | sudo tee /etc/yum.repos.d/mongodb-org-5.0.repo

# Install MongoDB
sudo yum install -y mongodb-org
sudo systemctl enable --now mongod

    Microsoft SQL Server

    Installation

    bash 
    # Add Microsoft repository
echo '[packages-microsoft-com-mssql-server-2019]
name=packages-microsoft-com-mssql-server-2019
baseurl=https://packages.microsoft.com/rhel/$releasever/mssql-server-2019/
enabled=1
gpgcheck=1
gpgkey=https://packages.microsoft.com/keys/microsoft.asc' | sudo tee /etc/yum.repos.d/mssql-server.repo

# Install SQL Server
sudo yum install -y mssql-server
sudo /opt/mssql/bin/mssql-conf setup
sudo systemctl enable --now mssql-server

    CouchDB

    Installation

    bash 
    # Add CouchDB repository
echo '[bintray--apache-couchdb-rpm]
name=bintray--apache-couchdb-rpm
baseurl=http://apache.bintray.com/couchdb-rpm/el$releasever/$basearch/
gpgcheck=0
repo_gpgcheck=0
enabled=1' | sudo tee /etc/yum.repos.d/apache-couchdb.repo

# Install CouchDB
sudo yum install -y couchdb
sudo systemctl enable --now couchdb

    Configuration

    Firewall Setup

    bash 
    # MariaDB/MySQL
sudo firewall-cmd --permanent --add-service=mysql
# PostgreSQL
sudo firewall-cmd --permanent --add-port=5432/tcp
# MongoDB
sudo firewall-cmd --permanent --add-port=27017/tcp
# SQL Server
sudo firewall-cmd --permanent --add-port=1433/tcp
# CouchDB
sudo firewall-cmd --permanent --add-port=5984/tcp

sudo firewall-cmd --reload

    Verification

    bash 
    # MariaDB/MySQL
mysql -u root -p -e "SELECT VERSION();"

# PostgreSQL
sudo -u postgres psql -c "SELECT version();"

# MongoDB
mongosh --eval "db.version()"

# SQL Server
sqlcmd -S localhost -U SA -P 'YourPassword' -Q 'SELECT @@VERSION'

# CouchDB
curl http://localhost:5984/

    Usage

    Each database system has different management approaches:

    1. MariaDB/MySQL: Use mysql command-line client or phpMyAdmin

    2. PostgreSQL: Use psql command-line client or pgAdmin

    3. MongoDB: Use mongosh shell or MongoDB Compass

    4. SQL Server: Use sqlcmd or SQL Server Management Studio

    5. CouchDB: Use web interface at http://localhost:5984/_utils

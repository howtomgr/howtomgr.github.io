Rocket.Chat Installation Guide

Open-source team communication platform with features like real-time messaging, video calls, file sharing, and extensive integrations.

Prerequisites

  • Linux system (Ubuntu/Debian, RHEL/CentOS, or compatible)
  • 4GB RAM minimum, 8GB recommended
  • MongoDB 4.4+ or 5.0+
  • Node.js 14.x
  • Domain name (for SSL setup)

    • Installation

    Using Docker Compose (Recommended)

    bash 
    # Create project directory
mkdir -p ~/rocketchat && cd ~/rocketchat

# Create docker-compose.yml
cat > docker-compose.yml <<EOF
version: '3.8'

services:
  rocketchat:
    image: registry.rocket.chat/rocketchat/rocket.chat:latest
    restart: unless-stopped
    environment:
      MONGO_URL: "mongodb://mongo:27017/rocketchat?replicaSet=rs0"
      MONGO_OPLOG_URL: "mongodb://mongo:27017/local?replicaSet=rs0"
      ROOT_URL: "https://chat.example.com"
      PORT: 3000
      DEPLOY_METHOD: docker
      REG_TOKEN: \${REG_TOKEN}
    depends_on:
      - mongo
    ports:
      - "3000:3000"
    volumes:
      - ./uploads:/app/uploads
    networks:
      - rocketchat

  mongo:
    image: docker.io/bitnami/mongodb:5.0
    restart: unless-stopped
    environment:
      MONGODB_REPLICA_SET_MODE: primary
      MONGODB_REPLICA_SET_NAME: rs0
      MONGODB_PORT_NUMBER: 27017
      MONGODB_INITIAL_PRIMARY_HOST: mongo
      MONGODB_INITIAL_PRIMARY_PORT_NUMBER: 27017
      MONGODB_ADVERTISED_HOSTNAME: mongo
      MONGODB_ENABLE_JOURNAL: true
      ALLOW_EMPTY_PASSWORD: yes
    volumes:
      - mongodb_data:/bitnami/mongodb
    networks:
      - rocketchat

  mongo-express:
    image: mongo-express
    restart: unless-stopped
    environment:
      ME_CONFIG_MONGODB_URL: "mongodb://mongo:27017/"
      ME_CONFIG_BASICAUTH_USERNAME: admin
      ME_CONFIG_BASICAUTH_PASSWORD: pass
    ports:
      - "8081:8081"
    depends_on:
      - mongo
    networks:
      - rocketchat

networks:
  rocketchat:
    driver: bridge

volumes:
  mongodb_data:
EOF

# Start services
docker-compose up -d

# Check logs
docker-compose logs -f rocketchat

    Manual Installation (Ubuntu/Debian)

    #### MongoDB Setup

    bash 
    # Install MongoDB
wget -qO - https://www.mongodb.org/static/pgp/server-5.0.asc | sudo apt-key add -
echo "deb [ arch=amd64,arm64 ] https://repo.mongodb.org/apt/ubuntu focal/mongodb-org/5.0 multiverse" | sudo tee /etc/apt/sources.list.d/mongodb-org-5.0.list

sudo apt update
sudo apt install -y mongodb-org

# Configure MongoDB as replica set
sudo systemctl enable --now mongod

# Initialize replica set
mongo --eval "rs.initiate({_id: 'rs0', members: [{_id: 0, host: 'localhost:27017'}]})"

# Create Rocket.Chat database user
mongo rocketchat --eval "db.createUser({user: 'rocketchat', pwd: 'secure_password', roles: [{role: 'readWrite', db: 'rocketchat'}]})"

    #### Node.js and Rocket.Chat Setup

    bash 
    # Install Node.js 14.x
curl -fsSL https://deb.nodesource.com/setup_14.x | sudo -E bash -
sudo apt install -y nodejs

# Create rocketchat user
sudo useradd -m -s /bin/bash rocketchat

# Download and install Rocket.Chat
cd /opt
sudo wget https://releases.rocket.chat/latest/download -O rocket.chat.tgz
sudo tar -xzf rocket.chat.tgz
sudo mv bundle Rocket.Chat
sudo chown -R rocketchat:rocketchat /opt/Rocket.Chat

# Install dependencies
cd /opt/Rocket.Chat/programs/server
sudo -u rocketchat npm install

# Set environment variables
sudo tee /etc/systemd/system/rocketchat.service > /dev/null <<EOF
[Unit]
Description=The Rocket.Chat server
After=network.target remote-fs.target nss-lookup.target nginx.service mongod.service
Wants=network.target

[Service]
ExecStart=/usr/local/bin/node /opt/Rocket.Chat/main.js
StandardOutput=syslog
StandardError=syslog
SyslogIdentifier=rocketchat
User=rocketchat
Group=rocketchat
Environment=MONGO_URL=mongodb://rocketchat:secure_password@localhost:27017/rocketchat?authSource=rocketchat&replicaSet=rs0
Environment=MONGO_OPLOG_URL=mongodb://rocketchat:secure_password@localhost:27017/local?authSource=admin&replicaSet=rs0
Environment=ROOT_URL=https://chat.example.com
Environment=PORT=3000

[Install]
WantedBy=multi-user.target
EOF

# Enable and start service
sudo systemctl daemon-reload
sudo systemctl enable --now rocketchat

    Configuration

    NGINX Reverse Proxy

    bash 
    # Create NGINX configuration
sudo tee /etc/nginx/sites-available/rocketchat > /dev/null <<EOF
# Upstreams
upstream backend {
    server 127.0.0.1:3000;
}

# HTTPS Server
server {
    listen 443 ssl http2;
    server_name chat.example.com;

    # SSL Configuration
    ssl_certificate /etc/letsencrypt/live/chat.example.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/chat.example.com/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN";
    add_header X-XSS-Protection "1; mode=block";
    add_header X-Content-Type-Options "nosniff";
    add_header Referrer-Policy "no-referrer-when-downgrade";
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # Proxy settings
    location / {
        proxy_pass http://backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade \$http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host \$http_host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
        proxy_set_header X-Nginx-Proxy true;

        proxy_redirect off;
    }
}

# HTTP to HTTPS redirect
server {
    listen 80;
    server_name chat.example.com;
    return 301 https://\$server_name\$request_uri;
}
EOF

# Enable site
sudo ln -s /etc/nginx/sites-available/rocketchat /etc/nginx/sites-enabled/
sudo nginx -t && sudo systemctl restart nginx

    Backup and Restore

    bash 
    # Create backup script
sudo tee /usr/local/bin/rocketchat-backup.sh > /dev/null <<'EOF'
#!/bin/bash
BACKUP_DIR="/backup/rocketchat"
DATE=$(date +%Y%m%d_%H%M%S)

mkdir -p ${BACKUP_DIR}/{database,uploads}

# Backup MongoDB database
mongodump --host localhost:27017 --db rocketchat --out ${BACKUP_DIR}/database/mongodb-${DATE}
tar -czf ${BACKUP_DIR}/database/rocketchat-db-${DATE}.tar.gz -C ${BACKUP_DIR}/database mongodb-${DATE}
rm -rf ${BACKUP_DIR}/database/mongodb-${DATE}

# Backup uploads directory
tar -czf ${BACKUP_DIR}/uploads/rocketchat-uploads-${DATE}.tar.gz -C /opt/Rocket.Chat uploads

# Docker backup (if using Docker)
if command -v docker &> /dev/null; then
    docker-compose exec mongo mongodump --archive=/tmp/rocketchat-backup.archive --db=rocketchat
    docker cp rocketchat_mongo_1:/tmp/rocketchat-backup.archive ${BACKUP_DIR}/database/rocketchat-docker-${DATE}.archive
fi

# Keep only last 7 backups
find ${BACKUP_DIR} -name "rocketchat-*" -type f -mtime +7 -delete

echo "Backup completed: rocketchat-${DATE}"
EOF

sudo chmod +x /usr/local/bin/rocketchat-backup.sh

    Security Configuration

    Firewall Setup

    bash 
    # UFW (Ubuntu/Debian)
sudo ufw allow 3000/tcp   # Rocket.Chat
sudo ufw allow 27017/tcp  # MongoDB (restrict to localhost in production)
sudo ufw allow 'Nginx Full'
sudo ufw enable

# Firewalld (RHEL/CentOS)
sudo firewall-cmd --permanent --add-port=3000/tcp
sudo firewall-cmd --permanent --add-service=http
sudo firewall-cmd --permanent --add-service=https
sudo firewall-cmd --reload

    Security Hardening

    bash 
    # MongoDB security
sudo tee -a /etc/mongod.conf > /dev/null <<EOF

# Enable authentication
security:
  authorization: enabled

# Bind to localhost only
net:
  bindIp: 127.0.0.1
EOF

# Create MongoDB admin user
mongo admin --eval "
db.createUser({
  user: 'admin',
  pwd: 'secure_password',
  roles: [{role: 'userAdminAnyDatabase', db: 'admin'}]
});

db.createUser({
  user: 'rocketchat',
  pwd: 'rocketchat_password',
  roles: [{role: 'readWrite', db: 'rocketchat'}]
});
"

sudo systemctl restart mongod

# Update Rocket.Chat MongoDB URL with authentication
sudo sed -i 's|MONGO_URL=.*|MONGO_URL=mongodb://rocketchat:rocketchat_password@localhost:27017/rocketchat?authSource=rocketchat&replicaSet=rs0|' /etc/systemd/system/rocketchat.service

    High Availability Setup

    Load Balancer Configuration

    bash 
    # HAProxy configuration for Rocket.Chat cluster
sudo tee -a /etc/haproxy/haproxy.cfg > /dev/null <<EOF

# Rocket.Chat frontend
frontend rocketchat_frontend
    bind *:80
    bind *:443 ssl crt /etc/ssl/certs/chat.example.com.pem
    redirect scheme https if !{ ssl_fc }
    default_backend rocketchat_servers

# Rocket.Chat backend
backend rocketchat_servers
    balance roundrobin
    option httpchk GET /api/info
    http-check expect status 200
    
    # WebSocket support
    timeout tunnel 3600s
    
    server chat1 192.168.1.10:3000 check
    server chat2 192.168.1.11:3000 check
    server chat3 192.168.1.12:3000 check
EOF

sudo systemctl restart haproxy

    MongoDB Replica Set (Multi-node)

    bash 
    # On each MongoDB node, configure replica set
sudo tee -a /etc/mongod.conf > /dev/null <<EOF
replication:
  replSetName: rs0
  
net:
  bindIp: 0.0.0.0  # Allow connections from other nodes
EOF

sudo systemctl restart mongod

# Initialize replica set (run on primary node only)
mongo --eval "
rs.initiate({
  _id: 'rs0',
  members: [
    {_id: 0, host: 'mongo1.example.com:27017', priority: 2},
    {_id: 1, host: 'mongo2.example.com:27017', priority: 1},
    {_id: 2, host: 'mongo3.example.com:27017', priority: 1, arbiterOnly: true}
  ]
});
"

# Update Rocket.Chat MongoDB URL for replica set
Environment=MONGO_URL=mongodb://mongo1.example.com:27017,mongo2.example.com:27017,mongo3.example.com:27017/rocketchat?replicaSet=rs0

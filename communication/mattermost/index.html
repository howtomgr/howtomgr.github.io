Mattermost Installation Guide

Open-source, self-hosted team communication platform with enterprise-grade security. Complete alternative to Slack with advanced features for secure collaboration, workflow automation, and integrated voice/video calling.

Prerequisites

  • Linux system (any modern distribution)
  • Root or sudo access
  • 4GB RAM minimum, 8GB+ recommended for production
  • PostgreSQL 11+ or MySQL 8.0+ database
  • NGINX reverse proxy (recommended for production)
  • Domain name and SSL certificate for HTTPS

    • Installation

    Ubuntu/Debian (Recommended)

    bash 
    # Update system packages
sudo apt update && sudo apt upgrade -y

# Download latest Mattermost server
MATTERMOST_VERSION="9.2.4"  # Check for latest at https://mattermost.com/download
wget https://releases.mattermost.com/${MATTERMOST_VERSION}/mattermost-${MATTERMOST_VERSION}-linux-amd64.tar.gz

# Extract and install
tar -xzf mattermost-${MATTERMOST_VERSION}-linux-amd64.tar.gz
sudo mv mattermost /opt/

# Create system user
sudo useradd --system --user-group mattermost
sudo chown -R mattermost:mattermost /opt/mattermost

# Create data directory
sudo mkdir -p /opt/mattermost/data
sudo chown mattermost:mattermost /opt/mattermost/data
sudo chmod 755 /opt/mattermost/data

# Set proper permissions
sudo find /opt/mattermost -type d -exec chmod 755 {} \;
sudo find /opt/mattermost -type f -exec chmod 644 {} \;
sudo chmod +x /opt/mattermost/bin/mattermost

# Create log directory
sudo mkdir -p /var/log/mattermost
sudo chown mattermost:mattermost /var/log/mattermost

    Database Configuration

    PostgreSQL Setup (Recommended)

    bash 
    # Install PostgreSQL 15
sudo apt install -y postgresql-15 postgresql-client-15

# Create Mattermost database and user
sudo -u postgres psql <<EOF
CREATE DATABASE mattermost WITH ENCODING 'UTF8' LC_COLLATE='en_US.UTF-8' LC_CTYPE='en_US.UTF-8' TEMPLATE=template0;
CREATE USER mattermost WITH ENCRYPTED PASSWORD 'MattermostDBPassword2024!';
GRANT ALL PRIVILEGES ON DATABASE mattermost TO mattermost;

-- Performance optimization for Mattermost
ALTER SYSTEM SET shared_buffers = '1GB';
ALTER SYSTEM SET effective_cache_size = '3GB';
ALTER SYSTEM SET work_mem = '16MB';
ALTER SYSTEM SET maintenance_work_mem = '256MB';
ALTER SYSTEM SET checkpoint_completion_target = 0.9;
ALTER SYSTEM SET wal_buffers = '16MB';
ALTER SYSTEM SET random_page_cost = 1.1;

-- Mattermost-specific settings
ALTER DATABASE mattermost SET default_text_search_config = 'pg_catalog.english';

SELECT pg_reload_conf();
EOF

# Configure PostgreSQL for Mattermost
sudo tee -a /etc/postgresql/15/main/postgresql.conf > /dev/null <<EOF

# Mattermost optimizations
max_connections = 300
shared_preload_libraries = 'pg_stat_statements'
track_activity_query_size = 2048
pg_stat_statements.track = all
EOF

sudo systemctl restart postgresql

    Security Hardening

    NGINX Reverse Proxy with Security

    bash 
    # Create secure NGINX configuration
sudo tee /etc/nginx/sites-available/mattermost > /dev/null <<EOF
# Mattermost NGINX Configuration with Security Headers

upstream mattermost {
    server 127.0.0.1:8065;
    keepalive 64;
}

proxy_cache_path /var/cache/nginx/mattermost levels=1:2 keys_zone=mattermost_cache:10m max_size=3g inactive=120m use_temp_path=off;

# HTTP to HTTPS redirect
server {
    listen 80;
    server_name mattermost.example.com;
    return 301 https://\$server_name\$request_uri;
}

# HTTPS server
server {
    listen 443 ssl http2;
    server_name mattermost.example.com;

    # SSL configuration
    ssl_certificate /etc/letsencrypt/live/mattermost.example.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/mattermost.example.com/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:50m;
    ssl_session_timeout 1d;
    ssl_session_tickets off;

    # Security headers
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Content-Type-Options nosniff always;
    add_header X-Frame-Options DENY always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Content-Security-Policy "frame-ancestors 'none'" always;

    # Rate limiting
    limit_req_zone \$binary_remote_addr zone=mattermost_ratelimit:10m rate=10r/s;
    limit_req zone=mattermost_ratelimit burst=20 nodelay;

    # Max upload size for file attachments
    client_max_body_size 50M;

    location ~ /api/v[0-9]+/(users/)?websocket\$ {
        proxy_set_header Upgrade \$http_upgrade;
        proxy_set_header Connection "upgrade";
        client_max_body_size 50M;
        proxy_set_header Host \$http_host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
        proxy_set_header X-Frame-Options SAMEORIGIN;
        proxy_buffers 256 16k;
        proxy_buffer_size 16k;
        client_body_timeout 60;
        send_timeout 300;
        lingering_timeout 5;
        proxy_connect_timeout 90;
        proxy_send_timeout 300;
        proxy_read_timeout 90s;
        proxy_pass http://mattermost;
    }

    location / {
        client_max_body_size 50M;
        proxy_set_header Connection "";
        proxy_set_header Host \$http_host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
        proxy_set_header X-Frame-Options SAMEORIGIN;
        proxy_buffers 256 16k;
        proxy_buffer_size 16k;
        proxy_read_timeout 600s;
        proxy_cache mattermost_cache;
        proxy_cache_revalidate on;
        proxy_cache_min_uses 2;
        proxy_cache_use_stale timeout;
        proxy_cache_lock on;
        proxy_http_version 1.1;
        proxy_pass http://mattermost;
    }
}
EOF

# Enable site and install SSL
sudo ln -s /etc/nginx/sites-available/mattermost /etc/nginx/sites-enabled/
sudo certbot --nginx -d mattermost.example.com
sudo nginx -t && sudo systemctl reload nginx

    Backup and Monitoring

    Backup Strategy

    bash 
    sudo tee /usr/local/bin/mattermost-backup.sh > /dev/null <<'EOF'
#!/bin/bash
BACKUP_DIR="/backup/mattermost"
DATE=$(date +%Y%m%d_%H%M%S)

mkdir -p ${BACKUP_DIR}/{database,config,data,plugins}

echo "Starting Mattermost backup..."

# Stop Mattermost service
systemctl stop mattermost

# Backup database
sudo -u postgres pg_dump mattermost > ${BACKUP_DIR}/database/mattermost-${DATE}.sql
gzip ${BACKUP_DIR}/database/mattermost-${DATE}.sql

# Backup configuration
cp /opt/mattermost/config/config.json ${BACKUP_DIR}/config/config-${DATE}.json

# Backup user data and files
tar -czf ${BACKUP_DIR}/data/mattermost-data-${DATE}.tar.gz -C /opt/mattermost data/

# Backup plugins
if [ -d "/opt/mattermost/plugins" ]; then
    tar -czf ${BACKUP_DIR}/plugins/mattermost-plugins-${DATE}.tar.gz -C /opt/mattermost plugins/
fi

# Start Mattermost
systemctl start mattermost

# Upload to cloud storage
aws s3 cp ${BACKUP_DIR}/ s3://mattermost-backups/ --recursive
gsutil cp -r ${BACKUP_DIR}/* gs://mattermost-backups/

# Keep last 7 backups
find ${BACKUP_DIR} -name "mattermost-*" -type f -mtime +7 -delete

echo "Mattermost backup completed: ${DATE}"
EOF

sudo chmod +x /usr/local/bin/mattermost-backup.sh
echo "0 2 * * * root /usr/local/bin/mattermost-backup.sh" | sudo tee -a /etc/crontab

    Health Monitoring

    bash 
    sudo tee /usr/local/bin/mattermost-health.sh > /dev/null <<'EOF'
#!/bin/bash
HEALTH_LOG="/var/log/mattermost-health.log"

log_message() {
    echo "$(date '+%Y-%m-%d %H:%M:%S') - $1" | tee -a ${HEALTH_LOG}
}

# Check service status
if systemctl is-active mattermost >/dev/null 2>&1; then
    log_message "✓ Mattermost service is running"
else
    log_message "✗ Mattermost service is not running"
    exit 1
fi

# Check web interface
HTTP_STATUS=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:8065/api/v4/system/ping)
if [ "${HTTP_STATUS}" = "200" ]; then
    log_message "✓ Mattermost API is responding"
else
    log_message "✗ Mattermost API issues (HTTP ${HTTP_STATUS})"
fi

# Check database connectivity
DB_STATUS=$(curl -s http://localhost:8065/api/v4/database/ping | jq -r '.database_status' 2>/dev/null)
if [ "$DB_STATUS" = "OK" ]; then
    log_message "✓ Database connection healthy"
else
    log_message "⚠ Database connection issues"
fi

# Check disk space
DISK_USAGE=$(df /opt/mattermost | awk 'NR==2 {print $5}' | sed 's/%//')
if [ ${DISK_USAGE} -gt 85 ]; then
    log_message "⚠ High disk usage: ${DISK_USAGE}%"
else
    log_message "✓ Disk usage: ${DISK_USAGE}%"
fi

log_message "Mattermost health check completed"
EOF

sudo chmod +x /usr/local/bin/mattermost-health.sh
echo "*/15 * * * * root /usr/local/bin/mattermost-health.sh" | sudo tee -a /etc/crontab

Grafana Installation Guide

Grafana is a free and open-source multi-platform analytics and interactive visualization web application. Originally developed by Torkel Ödegaard, Grafana provides charts, graphs, and alerts for the web when connected to supported data sources. It serves as a FOSS alternative to commercial solutions like Datadog, New Relic, or Splunk dashboards, offering enterprise-grade visualization and monitoring without licensing costs, with features like alerting, authentication integration, and extensive plugin ecosystem.

Prerequisites

  • Hardware Requirements:
  • CPU: 1 core minimum (2+ cores recommended for production)
  • RAM: 512MB minimum (2GB+ recommended for production)
  • Storage: 1GB minimum (10GB+ recommended for dashboard storage and logs)
  • Network: Stable internet connection for plugins and updates
  • Operating System:
  • Linux: Any modern distribution with systemd support
  • macOS: 10.15+ (Catalina or newer)
  • Windows: Windows Server 2016+ or Windows 10
  • FreeBSD: 12.0+
  • Network Requirements:
  • Port 3000 (default Grafana web interface)
  • Port 443/80 (HTTPS/HTTP if using reverse proxy)
  • Outbound HTTPS for plugin downloads and updates
  • Dependencies:
  • Database (SQLite, MySQL, PostgreSQL) - SQLite included by default
  • Data sources (Prometheus, InfluxDB, etc.) for meaningful dashboards
  • Web browser with JavaScript support
  • System Access: root or sudo privileges required

    • Installation

    RHEL/CentOS/Rocky Linux/AlmaLinux

    bash 
    # Add Grafana repository
sudo tee /etc/yum.repos.d/grafana.repo <<EOF
[grafana]
name=grafana
baseurl=https://packages.grafana.com/oss/rpm
repo_gpgcheck=1
enabled=1
gpgcheck=1
gpgkey=https://packages.grafana.com/gpg.key
sslverify=1
sslcacert=/etc/pki/tls/certs/ca-bundle.crt
EOF

# Install Grafana
sudo yum install -y grafana

# Enable and start service
sudo systemctl enable --now grafana-server

# Configure firewall
sudo firewall-cmd --permanent --add-port=3000/tcp
sudo firewall-cmd --reload

    Debian/Ubuntu

    bash 
    # Update package index and install prerequisites
sudo apt update
sudo apt install -y software-properties-common wget

# Add Grafana GPG key
wget -q -O - https://packages.grafana.com/gpg.key | sudo apt-key add -

# Add repository
echo "deb https://packages.grafana.com/oss/deb stable main" | sudo tee /etc/apt/sources.list.d/grafana.list

# Install Grafana
sudo apt update
sudo apt install -y grafana

# Enable and start service
sudo systemctl enable --now grafana-server

# Configure firewall
sudo ufw allow 3000

    First-Run Setup

    1. Create grafana user (if not created by package):

    2. Default configuration locations:

  • RHEL/CentOS/Rocky/AlmaLinux: /etc/grafana/grafana.ini

server {
    listen 443 ssl http2;
    server_name grafana.example.com;

    ssl_certificate /etc/letsencrypt/live/grafana.example.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/grafana.example.com/privkey.pem;

    # Security headers
    add_header Strict-Transport-Security "max-age=63072000" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;

    location / {
        proxy_pass http://grafana;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $http_host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}

    Backup and Restore

    Backup Procedures

    bash 
    #!/bin/bash
# backup-grafana.sh

BACKUP_DIR="/backup/grafana/$(date +%Y%m%d_%H%M%S)"
mkdir -p "$BACKUP_DIR"

# Stop Grafana for consistent backup
sudo systemctl stop grafana-server

# Backup SQLite database (if using SQLite)
if [ -f /var/lib/grafana/grafana.db ]; then
    cp /var/lib/grafana/grafana.db "$BACKUP_DIR/grafana.db"
fi

# Backup MySQL database (if using MySQL)
if grep -q "type = mysql" /etc/grafana/grafana.ini; then
    mysqldump -u grafana -p grafana > "$BACKUP_DIR/grafana-mysql.sql"
fi

# Backup PostgreSQL database (if using PostgreSQL)
if grep -q "type = postgres" /etc/grafana/grafana.ini; then
    pg_dump -U grafana grafana > "$BACKUP_DIR/grafana-postgres.sql"
fi

# Backup configuration
tar czf "$BACKUP_DIR/grafana-config.tar.gz" /etc/grafana/

# Backup plugins
tar czf "$BACKUP_DIR/grafana-plugins.tar.gz" /var/lib/grafana/plugins/

# Backup dashboards via API
mkdir -p "$BACKUP_DIR/dashboards"
curl -s "http://admin:password@localhost:3000/api/search?type=dash-db" | \
    jq -r '.[] | .uid' | while read uid; do
        curl -s "http://admin:password@localhost:3000/api/dashboards/uid/$uid" > \
            "$BACKUP_DIR/dashboards/$uid.json"
    done

# Start Grafana
sudo systemctl start grafana-server

echo "Backup completed: $BACKUP_DIR"

    Restore Procedures

    bash 
    #!/bin/bash
# restore-grafana.sh

BACKUP_DIR="$1"
if [ -z "$BACKUP_DIR" ]; then
    echo "Usage: $0 <backup-directory>"
    exit 1
fi

# Stop Grafana
sudo systemctl stop grafana-server

# Restore database
if [ -f "$BACKUP_DIR/grafana.db" ]; then
    cp "$BACKUP_DIR/grafana.db" /var/lib/grafana/grafana.db
    chown grafana:grafana /var/lib/grafana/grafana.db
fi

# Restore MySQL database
if [ -f "$BACKUP_DIR/grafana-mysql.sql" ]; then
    mysql -u grafana -p grafana < "$BACKUP_DIR/grafana-mysql.sql"
fi

# Restore PostgreSQL database
if [ -f "$BACKUP_DIR/grafana-postgres.sql" ]; then
    dropdb grafana && createdb grafana
    psql -U grafana grafana < "$BACKUP_DIR/grafana-postgres.sql"
fi

# Restore configuration
tar xzf "$BACKUP_DIR/grafana-config.tar.gz" -C /

# Restore plugins
tar xzf "$BACKUP_DIR/grafana-plugins.tar.gz" -C /

# Fix ownership
sudo chown -R grafana:grafana /var/lib/grafana /etc/grafana

# Start Grafana
sudo systemctl start grafana-server

echo "Restore completed from: $BACKUP_DIR"

    Troubleshooting

    Common Issues

    1. Cannot connect to Grafana web interface:

    bash 
    # Check service status
sudo systemctl status grafana-server

# Check logs
sudo journalctl -u grafana-server -f

# Check port binding
sudo ss -tlnp | grep :3000

# Test local connection
curl -I http://localhost:3000

    2. Database connection issues:

    bash 
    # Test database connectivity
# MySQL
mysql -h localhost -u grafana -p grafana -e "SELECT 1;"

# PostgreSQL
psql -h localhost -U grafana -d grafana -c "SELECT 1;"

# Check database logs
sudo tail -f /var/log/mysql/error.log
sudo tail -f /var/log/postgresql/postgresql-*.log

    3. Plugin installation failures:

    bash 
    # Check plugin directory permissions
ls -la /var/lib/grafana/plugins/

# Install plugin manually
sudo grafana-cli --pluginUrl https://example.com/plugin.zip plugins install custom-plugin

# Check plugin logs
sudo journalctl -u grafana-server | grep -i plugin

    4. High memory usage:

    bash 
    # Check Grafana memory usage
ps aux | grep grafana-server

# Monitor memory growth
top -p $(pgrep grafana-server)

# Check for memory leaks in logs
sudo grep -i "out of memory" /var/log/grafana/grafana.log

# Restart if needed
sudo systemctl restart grafana-server

    Health Checks

    bash 
    #!/bin/bash
# grafana-health-check.sh

# Check service status
if ! systemctl is-active grafana-server >/dev/null 2>&1; then
    echo "CRITICAL: Grafana service is not running"
    exit 2
fi

# Check web interface
if ! curl -f -s http://localhost:3000/api/health >/dev/null; then
    echo "CRITICAL: Grafana web interface is not responding"
    exit 2
fi

# Check database connectivity
if ! curl -s http://localhost:3000/api/health | grep -q '"database":"ok"'; then
    echo "WARNING: Database connectivity issues"
    exit 1
fi

# Check disk space
DISK_USAGE=$(df /var/lib/grafana | awk 'NR==2 {print $5}' | sed 's/%//')
if [ "$DISK_USAGE" -gt 85 ]; then
    echo "WARNING: Disk usage is ${DISK_USAGE}%"
    exit 1
fi

echo "OK: Grafana is healthy"
exit 0

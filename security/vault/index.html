HashiCorp Vault Installation Guide

Enterprise-grade secrets management platform for securely storing and accessing API keys, passwords, certificates, and other sensitive data. Industry standard for secrets management with advanced security features.

Prerequisites

  • Linux system (any modern distribution)
  • Root or sudo access
  • 4GB RAM minimum, 8GB+ recommended for production
  • Persistent storage for Vault data (encrypted disk recommended)
  • TLS certificates for production deployment
  • HSM (optional, for enterprise compliance)

    • Installation

    Using Package Manager (Recommended)

    #### Ubuntu/Debian

    bash 
    # Add HashiCorp GPG key
wget -O- https://apt.releases.hashicorp.com/gpg | sudo gpg --dearmor -o /usr/share/keyrings/hashicorp-archive-keyring.gpg

# Add HashiCorp repository
echo "deb [signed-by=/usr/share/keyrings/hashicorp-archive-keyring.gpg] https://apt.releases.hashicorp.com $(lsb_release -cs) main" | sudo tee /etc/apt/sources.list.d/hashicorp.list

# Update and install Vault
sudo apt update
sudo apt install -y vault

# Create vault user and directories
sudo useradd --system --home /etc/vault.d --shell /bin/false vault
sudo mkdir -p /opt/vault/{data,logs,tls} /etc/vault.d
sudo chown -R vault:vault /opt/vault /etc/vault.d

    bash

# Clean up
rm vault_${VAULT_VERSION}_linux_amd64.zip

    Using Docker

    bash 
    # Create directories for persistent data
mkdir -p ~/vault/{data,config,logs,tls}

# Generate TLS certificates for Docker
openssl req -x509 -nodes -newkey rsa:4096 -keyout ~/vault/tls/vault.key -out ~/vault/tls/vault.crt -days 365 -subj "/C=US/ST=State/L=City/O=Organization/CN=vault.example.com"

# Create production-ready configuration
cat > ~/vault/config/vault.hcl <<EOF
ui = true
disable_mlock = false

# Storage backend
storage "file" {
  path = "/vault/data"
}

# HTTPS listener
listener "tcp" {
  address       = "0.0.0.0:8200"
  tls_cert_file = "/vault/tls/vault.crt"
  tls_key_file  = "/vault/tls/vault.key"
  tls_min_version = "tls12"
}

# Cluster configuration
api_addr = "https://vault.example.com:8200"
cluster_addr = "https://vault.example.com:8201"

# Logging
log_level = "Info"
log_file  = "/vault/logs/vault.log"
log_rotate_duration = "24h"
log_rotate_max_files = 15

# Telemetry
telemetry {
  prometheus_retention_time = "30s"
  disable_hostname = true
}

# Enterprise features (if using Vault Enterprise)
license_path = "/vault/license/vault.hclic"

# Performance replication (Enterprise)
replication {
  resolver_discover_servers = false
}
EOF

# Run Vault container with security settings
docker run -d \
  --name vault \
  --restart unless-stopped \
  --cap-add=IPC_LOCK \
  --ulimit memlock=-1:-1 \
  -p 8200:8200 \
  -v ~/vault/data:/vault/data \
  -v ~/vault/config:/vault/config \
  -v ~/vault/logs:/vault/logs \
  -v ~/vault/tls:/vault/tls \
  vault:latest server -config=/vault/config/vault.hcl

    Advanced TLS Configuration

    bash 
    # Generate production-grade TLS certificates
sudo mkdir -p /opt/vault/tls

# Create CA certificate
sudo openssl genrsa -out /opt/vault/tls/ca-key.pem 4096
sudo openssl req -new -x509 -days 3650 -key /opt/vault/tls/ca-key.pem -out /opt/vault/tls/ca.crt -subj "/C=US/ST=State/L=City/O=Organization/CN=Vault-CA"

# Create server certificate
sudo openssl genrsa -out /opt/vault/tls/vault-key.pem 4096
sudo openssl req -new -key /opt/vault/tls/vault-key.pem -out /opt/vault/tls/vault.csr -subj "/C=US/ST=State/L=City/O=Organization/CN=vault.example.com"

# Create certificate with SAN extensions
sudo tee /opt/vault/tls/vault.conf > /dev/null <<EOF
[req]
distinguished_name = req_distinguished_name
req_extensions = v3_req

[req_distinguished_name]

[v3_req]
basicConstraints = CA:FALSE
keyUsage = nonRepudiation, digitalSignature, keyEncipherment
subjectAltName = @alt_names

[alt_names]
DNS.1 = vault.example.com
DNS.2 = vault-node-1.example.com
DNS.3 = vault-node-2.example.com
DNS.4 = vault-node-3.example.com
DNS.5 = localhost
IP.1 = 127.0.0.1
IP.2 = 10.0.1.10
IP.3 = 10.0.1.11
IP.4 = 10.0.1.12
EOF

sudo openssl x509 -req -days 365 -in /opt/vault/tls/vault.csr -CA /opt/vault/tls/ca.crt -CAkey /opt/vault/tls/ca-key.pem -CAcreateserial -out /opt/vault/tls/vault.crt -extensions v3_req -extfile /opt/vault/tls/vault.conf

# Set secure permissions
sudo chmod 600 /opt/vault/tls/*-key.pem
sudo chmod 644 /opt/vault/tls/*.crt
sudo chown -R vault:vault /opt/vault/tls

# For Let's Encrypt certificates (production)
sudo apt install -y certbot  # Ubuntu/Debian
sudo yum install -y certbot  # RHEL/CentOS

sudo certbot certonly --standalone -d vault.example.com
sudo cp /etc/letsencrypt/live/vault.example.com/fullchain.pem /opt/vault/tls/vault.crt
sudo cp /etc/letsencrypt/live/vault.example.com/privkey.pem /opt/vault/tls/vault-key.pem
sudo chown vault:vault /opt/vault/tls/vault.*

    Security Hardening (2024 Best Practices)

    System-Level Security

    bash 
    # Disable swap (critical for security)
sudo swapoff -a
sudo sed -i '/ swap / s/^\(.*\)$/#\1/g' /etc/fstab

# Disable core dumps
echo '* hard core 0' | sudo tee -a /etc/security/limits.conf
echo '* soft core 0' | sudo tee -a /etc/security/limits.conf
echo 'vault hard core 0' | sudo tee -a /etc/security/limits.conf
echo 'vault soft core 0' | sudo tee -a /etc/security/limits.conf

# Configure kernel parameters for security
sudo tee -a /etc/sysctl.conf > /dev/null <<EOF
# Vault security hardening
kernel.dmesg_restrict = 1
kernel.kptr_restrict = 2
kernel.yama.ptrace_scope = 1
fs.suid_dumpable = 0
net.ipv4.conf.all.send_redirects = 0
net.ipv4.conf.default.send_redirects = 0
net.ipv4.conf.all.accept_source_route = 0
net.ipv4.conf.default.accept_source_route = 0
EOF

sudo sysctl -p

# SELinux configuration (RHEL/CentOS/Fedora)
if command -v getenforce >/dev/null 2>&1; then
    sudo setsebool -P authlogin_nsswitch_use_ldap 1
    sudo setsebool -P allow_execheap 0
    sudo setsebool -P allow_execmem 0
    sudo setsebool -P allow_execmod 0
    sudo setsebool -P allow_execstack 0
    
    # Custom SELinux policy for Vault
    sudo tee vault.te > /dev/null <<EOF
module vault 1.0;

require {
    type unconfined_t;
    type vault_t;
    class capability { ipc_lock sys_admin };
    class process { setrlimit };
}

#============= vault_t ==============
allow vault_t self:capability { ipc_lock sys_admin };
allow vault_t self:process setrlimit;
EOF

    checkmodule -M -m -o vault.mod vault.te
    semodule_package -o vault.pp -m vault.mod
    sudo semodule -i vault.pp
fi

fi

# Check cluster status (if HA)
if vault status -format=json | jq -r .ha_enabled | grep -q "true"; then
    CLUSTER_HEALTH=$(vault operator raft list-peers -format=json | jq '.data.config.servers | length')
    log_message "‚úì Cluster has ${CLUSTER_HEALTH} nodes"
fi

# Check replication status (Enterprise)
REPLICATION_STATUS=$(vault read -format=json sys/replication/status 2>/dev/null)
if [ $? -eq 0 ]; then
    DR_MODE=$(echo $REPLICATION_STATUS | jq -r .data.dr.mode)
    PERF_MODE=$(echo $REPLICATION_STATUS | jq -r .data.performance.mode)
    log_message "‚úì Replication status - DR: ${DR_MODE}, Performance: ${PERF_MODE}"
fi

# Check auth methods
AUTH_METHODS=$(vault auth list -format=json | jq -r 'keys | length')
log_message "‚úì ${AUTH_METHODS} authentication methods enabled"

# Check secrets engines
SECRETS_ENGINES=$(vault secrets list -format=json | jq -r 'keys | length')
log_message "‚úì ${SECRETS_ENGINES} secrets engines enabled"

# Check policy count
POLICIES=$(vault policy list | wc -l)
log_message "‚úì ${POLICIES} policies configured"

# Test secret read/write
TEST_SECRET="secret/health-check/test-$(date +%s)"
if vault kv put ${TEST_SECRET} test=true >/dev/null 2>&1; then
    if vault kv get ${TEST_SECRET} >/dev/null 2>&1; then
        log_message "‚úì Secret read/write test passed"
        vault kv delete ${TEST_SECRET} >/dev/null 2>&1
    else
        log_message "‚úó Secret read test failed"
    fi
else
    log_message "‚úó Secret write test failed"
fi

# Check audit devices
AUDIT_DEVICES=$(vault audit list -format=json | jq -r 'keys | length')
log_message "‚úì ${AUDIT_DEVICES} audit devices enabled"

log_message "Vault health check completed successfully"
EOF

sudo chmod +x /usr/local/bin/vault-health-check.sh

# Schedule health checks every 15 minutes
echo "*/15 * * * * root /usr/local/bin/vault-health-check.sh" | sudo tee -a /etc/crontab

    Security Validation Testing

    bash 
    # Create security testing script
sudo tee /usr/local/bin/vault-security-test.sh > /dev/null <<'EOF'
#!/bin/bash
VAULT_ADDR="https://vault.example.com:8200"
TEST_RESULTS="/tmp/vault-security-test-$(date +%Y%m%d_%H%M%S).txt"

echo "Vault Security Validation Test" > ${TEST_RESULTS}
echo "==============================" >> ${TEST_RESULTS}
echo "Date: $(date)" >> ${TEST_RESULTS}
echo "" >> ${TEST_RESULTS}

# Test 1: TLS Configuration
echo "Testing TLS Configuration..." >> ${TEST_RESULTS}
TLS_VERSION=$(openssl s_client -connect vault.example.com:8200 -brief 2>/dev/null | grep "Protocol version")
echo "TLS Version: ${TLS_VERSION}" >> ${TEST_RESULTS}

# Test 2: Authentication requirements
echo "Testing authentication requirements..." >> ${TEST_RESULTS}
UNAUTH_ACCESS=$(curl -k -s -w "%{http_code}" -o /dev/null ${VAULT_ADDR}/v1/secret/test)
if [ "$UNAUTH_ACCESS" = "403" ]; then
    echo "‚úì Unauthenticated access properly denied" >> ${TEST_RESULTS}
else
    echo "‚úó Unauthenticated access not properly restricted" >> ${TEST_RESULTS}
fi

# Test 3: Seal status
echo "Testing seal status..." >> ${TEST_RESULTS}
SEALED=$(vault status -format=json | jq -r .sealed)
if [ "$SEALED" = "false" ]; then
    echo "‚úì Vault is properly unsealed" >> ${TEST_RESULTS}
else
    echo "‚úó Vault is sealed" >> ${TEST_RESULTS}
fi

# Test 4: Audit logging
echo "Testing audit logging..." >> ${TEST_RESULTS}
AUDIT_COUNT=$(vault audit list -format=json | jq -r 'keys | length')
if [ "$AUDIT_COUNT" -gt "0" ]; then
    echo "‚úì ${AUDIT_COUNT} audit devices enabled" >> ${TEST_RESULTS}
else
    echo "‚úó No audit devices enabled" >> ${TEST_RESULTS}
fi

# Test 5: Policy enforcement
echo "Testing policy enforcement..." >> ${TEST_RESULTS}
vault token create -policy=developer -ttl=1m -format=json > /tmp/test-token.json
TEST_TOKEN=$(jq -r .auth.client_token /tmp/test-token.json)

# Test with limited token
VAULT_TOKEN=${TEST_TOKEN} vault auth -method=token ${TEST_TOKEN} >/dev/null 2>&1
LIMITED_ACCESS=$(VAULT_TOKEN=${TEST_TOKEN} curl -k -s -w "%{http_code}" -o /dev/null -H "X-Vault-Token: ${TEST_TOKEN}" ${VAULT_ADDR}/v1/sys/auth)
if [ "$LIMITED_ACCESS" = "403" ]; then
    echo "‚úì Policy enforcement working correctly" >> ${TEST_RESULTS}
else
    echo "‚úó Policy enforcement may be misconfigured" >> ${TEST_RESULTS}
fi

# Revoke test token
vault token revoke ${TEST_TOKEN}
rm /tmp/test-token.json

echo "" >> ${TEST_RESULTS}
echo "Security test completed at: $(date)" >> ${TEST_RESULTS}

echo "Security test completed. Results: ${TEST_RESULTS}"
EOF

sudo chmod +x /usr/local/bin/vault-security-test.sh

# Run security tests weekly
echo "0 8 * * 1 root /usr/local/bin/vault-security-test.sh" | sudo tee -a /etc/crontab
